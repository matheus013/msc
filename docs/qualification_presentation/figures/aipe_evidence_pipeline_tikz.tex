\providecommand{\AipePipelineNode}[5]{%
  \ifnum#1=0
    \node[pipelineStep,#4] (#3) {#5};
  \else
    \ifnum#2=#1
      \node[activeStep,#4] (#3) {#5};
    \else
      \ifnum#2<#1
        \node[doneStep,#4] (#3) {#5};
      \else
        \node[futureStep,#4] (#3) {#5};
      \fi
    \fi
  \fi
}

\providecommand{\AipePipelineTag}[3]{%
  \ifnum#1=0
    \node[tag,#3] {#2};
  \else
    \ifnum#2=#1
      \node[tagActive,#3] {#2};
    \else
      \ifnum#2<#1
        \node[tagDone,#3] {#2};
      \else
        \node[tagFuture,#3] {#2};
      \fi
    \fi
  \fi
}

\providecommand{\AipePipelineFlow}[4]{%
  \ifnum#1=0
    \draw[flow] (#3) -- (#4);
  \else
    \ifnum#2<#1
      \draw[flowDone] (#3) -- (#4);
    \else
      \ifnum#2=#1
        \draw[flowActive] (#3) -- (#4);
      \else
        \draw[flowFuture] (#3) -- (#4);
      \fi
    \fi
  \fi
}
