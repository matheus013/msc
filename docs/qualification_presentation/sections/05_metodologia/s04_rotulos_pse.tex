% Slide — PSE: entrada, saída e estágio atual
\begin{frame}{Do rótulo ao PSE}
  \keymessage{O PSE aprende os rótulos gerados pelo benchmark e recomenda uma política para novas séries loja-produto.}
  \vspace{0.3em}

  \begin{columns}[T,onlytextwidth]
    \column{0.52\textwidth}
      \textbf{Treinamento do PSE}
      \begin{itemize}
        \footnotesize
        \item Entrada: $\mathbf{x}_i$, características operacionais da série
        \item Alvo supervisionado: $y_i$, política viável de menor CTI sob $\mathrm{NS} \geq 0{,}70$ (rótulo gerado pelo benchmark)
        \item Saída: $\hat{\pi}_i$, política recomendada, confiança e \textit{ranking}
      \end{itemize}
      {\footnotesize Meta-modelo supervisionado de seleção (algoritmo agnóstico nesta etapa).}
    \column{0.48\textwidth}
      \begin{alertblock}{Estágio atual}
        \footnotesize A regra de rotulagem e o pipeline estão definidos e testados preliminarmente. A validação em escala nacional do PSE é etapa de continuidade.
      \end{alertblock}
  \end{columns}
  \note{
    O PSE entra depois da rotulagem, já apresentada na formulação
    matemática do AIPE. Ele aprende a relação entre características
    operacionais $\mathbf{x}_i$ e o rótulo $y_i$. Na qualificação, a
    formulação está pronta e com testes preliminares. A validação em escala
    nacional permanece como etapa de continuidade.
  }
\end{frame}
