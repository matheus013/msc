% Metodologia -- Módulo 4 -- PODs
\begin{frame}{PODs: refinamento operacional dentro dos regimes}
  \scriptsize\textbf{\textcolor{darkblue}{A taxonomia Syntetos-Boylan classifica o regime macro da demanda. Os PODs refinam esse regime usando características adicionais da série loja-produto.}}
  \vspace{0.25em}

  \centering
  \tabstyle
  \scriptsize
  \begin{tabular}{@{}p{0.32\linewidth}p{0.58\linewidth}@{}}
    \toprule
    \textbf{POD} & \textbf{Critério ou grupo de características} \\
    \midrule
    Sparse High Impact & Longos silêncios + picos \\
    High Vol. Seasonal & Concentração/sazonalidade \\
    Unstable Trend & Tendência ou mudança de regime \\
    Low Vol. Stable & Baixo volume e baixa variabilidade \\
    Fast Moving & Demanda frequente \\
    \bottomrule
  \end{tabular}
  \vspace{0.25em}

  \scriptsize \fonte{capítulo de Metodologia, Tabela 4.1; demand\_profiling}\\[-0.05em]
  \scriptsize\textcolor{slategray}{Regime $\neq$ POD. No recorte BA, todas as séries são Lumpy, mas são separadas em PODs distintos para apoiar a seleção contextual.}
  \note{
    Cada série loja-produto é caracterizada e classificada em um POD; o
    POD não é a série, mas o perfil operacional atribuído a ela. A taxonomia
    de Syntetos-Boylan usa ADI e CV$^2$ para definir o regime macro. Os PODs
    refinam esse regime com características adicionais, como burstiness,
    tendência, sazonalidade e volume. No Experimento 2, todas as séries são
    Lumpy, mas os PODs separam subperfis operacionais dentro desse regime.
    O POD organiza o contexto; a recomendação de política vem depois, com a
    avaliação comparativa em simulação, a geração de rótulos e o PSE.
  }
\end{frame}
