% Slide 8 — Contribuição da dissertação
\begin{frame}{Contribuição da dissertação}
  \begin{enumerate}
    \item \textbf{Comparação empírica} de 12 políticas sobre dados reais de
      varejo brasileiro intermitente.
    \item \textbf{Mapeamento perfil~$\rightarrow$~política dominante}, com
      validação estatística formal.
    \item \textbf{Pipeline reprodutível} (dados reais, rastreabilidade
      completa de código e configuração).
  \end{enumerate}
  \vfill
  \begin{alertblock}{O que esta dissertação \textit{não} é}
    \small Não cria novos algoritmos de otimização ou de aprendizado por
    reforço. A contribuição está na \highlight{formulação e avaliação} do
    AIPE como \textit{framework} de seleção contextual.
  \end{alertblock}
  \note{
    É importante deixar claro o que é e o que não é a contribuição. A
    dissertação não propõe um novo algoritmo de GA, de RL ou de
    meta-heurística -- esses métodos já existem na literatura e são usados
    aqui como parte do portfólio avaliado. A contribuição real está em três
    frentes: a comparação empírica sistemática, o mapeamento entre perfil
    operacional e política dominante com suporte estatístico, e a
    infraestrutura reprodutível que sustenta as duas primeiras. O AIPE é a
    proposta central; as políticas individuais são meios, não fins.
  }
\end{frame}
